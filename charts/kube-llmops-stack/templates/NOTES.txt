{{- if .Values.gateway.enabled }}

1. Access the AI Gateway (LiteLLM):
   kubectl port-forward svc/{{ include "kube-llmops.fullname" . }}-litellm 4000:4000

   Then try:
   curl http://localhost:4000/v1/models -H "Authorization: Bearer {{ .Values.gateway.masterKey }}"

   kubectl port-forward svc/{{ include "kube-llmops.fullname" . }}-grafana 3000:3000
   Open http://localhost:3000 (admin / {{ .Values.observability.grafana.adminPassword | default "admin" }})

{{- end }}
